\section{算法介绍}

\subsection{一个好的训练过程}
\begin{algorithm}[!htp]
  \caption{训练过程}\label{alg_train}
  \begin{algorithmic}[1]
  \REQUIRE 特征 $a$,\par 
           矩阵 $adj$,\par 
           矩阵 $adj\_n$                  %输入条件
  \ENSURE 输出 $l\_adj$     %输出
  \STATE \textbf{begin}
      \STATE {好}
      \STATE {$l\_a$ $\leftarrow$ 映射函数($a$,$adj\_n$)}
      \STATE {好}
      \STATE {$l\_adj$ $\leftarrow$ 映射函数($adj$，$l\_a$)}
  
  \RETURN {$l\_adj$}
  \STATE \textbf{end}
  \end{algorithmic}
  \end{algorithm}
